获取当前文件的绝对路径

package main

import (
    "fmt"
    "path/filepath"
)

func main() {
    s, _ := filepath.Abs(".")
    fmt.Println("获取当前文件的绝对路径 : ", s) // d:\笔记\code
}

获取当前文件的上一级目录

package main

import (
    "fmt"
    "path/filepath"
)

func main() {
    s, _ := filepath.Abs(".")
    s2 := filepath.Base(s)
    fmt.Println("获取当前文件的上一级目录 : ", s2) // code
}

获取上一级目录的父目录

package main

import (
    "fmt"
    "path/filepath"
)

func main() {
    s, _ := filepath.Abs(".")
    s3 := filepath.Dir(s)
    fmt.Println("获取当前文件的父目录 : ", s3) // d:\笔记
}

清除不必要的字符

package main

import (
    "fmt"
    "path/filepath"
)

func main() {
    s4 := "../../...///.//../../../abc"
    fmt.Println("处理前的路径 : ", s4) // ../../...///.//../../../abc
    s4 = filepath.Clean(s4)
    fmt.Println("处理后的路径 : ", s4) // ..\..\..\..\abc

}

获取文件后缀

**方法一 : **

package main

import (
    "fmt"
    "path/filepath"
)

func Filepathmain() {
    s, _ := filepath.Abs("./file.go")

    s5 := filepath.Ext(s)
    fmt.Println(s5) // .go
}

**方法二 : **

package main

import (
    "fmt"
    "path/filepath"
    "strings"
)

func fileExt(path string) string {
    i := strings.LastIndex(path, ".")
    if i < 0 {
        return ""
    }
    return path[i:]
}

func main() {
    s, _ := filepath.Abs("./file.go")

    i := fileExt(s)
    fmt.Println("获取文件后缀", i) // 获取文件后缀 .go
}

判断路径是否被包含

package main

import (
    "fmt"
    "path/filepath"
)


func main() {
    s, _ := filepath.Abs("./file.go")
    s6 := filepath.Dir(s)

    b := strings.HasPrefix(s, s6) // 判断s6是否存在于s中
    fmt.Println(b)
}

路径拼接

package main

import (
    "fmt"
    "path/filepath"
)

func main() {
    s, _ := filepath.Abs("./file.go")
    s3 := filepath.Dir(s)

    s7 := filepath.Join(s3, "test.go")
    fmt.Println("拼接后的路径 : ", s7) // d:\笔记\code\test.go
}

文件和文件夹路径分割

package main

import (
    "fmt"
    "path/filepath"
)

func main() {
    s, _ := filepath.Abs("./file.go")

    dir, s8 := filepath.Split(s)
    fmt.Printf("分割好的文件夹 : %v, 分割好的文件名 : %v\n", dir, s8) // 分割好的文件夹 : d:\笔记\code\, 分割好的文件名 : file.go
}

环境变量分割

package main

import (
    "fmt"
    "path/filepath"
)


func main() {
    s := "c://1.txt;c://2.txt;d://123.txt"

    s9 := filepath.SplitList(s)
    fmt.Println(s9) // [c://1.txt c://2.txt d://123.txt]

}

检测文件夹下的指定后缀文件

package main

import (
    "fmt"
    "path/filepath"
)


func main() {
    s, _ = filepath.Abs(".")
    matches, _ := filepath.Glob(filepath.Join(s, "*.go"))
    fmt.Println(matches) // [d:\笔记\code\demo1\file.go d:\笔记\code\demo1\filepath_abs.go]
}
