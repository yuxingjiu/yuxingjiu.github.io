sync

sync.Mutex

设置互斥锁

var locker sync.Mutex

locker.Lock() // 上锁
locker.Unlock() // 解锁

sync.WaitGroup

WaitGroup实现同步

var wg sync.WaitGroup

wg.Add(n) // 添加n信号

wg.Done() // 处理完成一个信号

wg.Wait() // 等待计数器归零，当所有信号完成处理就结束

package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    var (
        locker sync.Mutex
        wg     sync.WaitGroup
    )

    wg.Add(2)

    go func() {
        fmt.Println("A : start")
        locker.Lock()
        fmt.Println("A : lock")
        time.Sleep(time.Second * 5)
        fmt.Println("A : unlock")
        fmt.Println("A : end")
        locker.Unlock()
        wg.Done()
    }()

    go func() {
        fmt.Println("B : start")
        locker.Lock()
        fmt.Println("B : lock")
        time.Sleep(time.Second * 5)
        locker.Unlock()
        fmt.Println("B : unlock")
        fmt.Println("B : end")

        wg.Done()
    }()

    wg.Wait()
}
/*
B : start
B : lock 
A : start
B : unlock
B : end 
A : lock
A : unlock
A : end
*/

sync.RWMutex

设置读写锁，也可以加锁和解锁，应用在同时读写的情况下

var locker sync.Mutex

locker.Lock() // 上写锁
locker.RLock() // 上读锁

locker.Unlock() // 解写锁
locker.RUnlock() // 解读锁

由于上锁的前提是会对文件造成修改，导致文件内容发生改变，才会选择加锁，但如果是多个例程同时去读取一个文件，并不会对文件内容造成修改，所以可以选择添加只读锁 ( RLock() ) ，而如果想要对文件进行修改，可以选择加可写锁 ( Lock() )

go func() {
    fmt.Println("A : start")
    locker.RLock()
    fmt.Println("A : lock")
    time.Sleep(time.Second * 5)
    fmt.Println("A : unlock")
    fmt.Println("A : end")
    locker.RUnlock()
    wg.Done()
}()
go func() {
    fmt.Println("B : start")
    locker.RLock()
    fmt.Println("B : lock")
    time.Sleep(time.Second * 5)
    locker.RUnlock()
    fmt.Println("B : unlock")
    fmt.Println("B : end")
}()
/*
B : start
B : lock 
A : start
A : lock 
A : unlock
A : end   
B : unlock
B : end   
*/

两个只读锁并不会互斥

Lock & Lock 互斥 （写 和 写）

RLock & Lock 互斥 （读 和 写）

RLock & RLock 不互斥（读 和 读）

sync.Cond

条件等待，通过sync.NewCond()创建

package main

import (
    "fmt"
    "sync"
    "time"
)

func main() {
    var (
        locker sync.Mutex
        wg     sync.WaitGroup
    )

    cond := sync.NewCond(&locker)

    wg.Add(2)

    go func() {
        cond.L.Lock()
        fmt.Println("cond wait")
        cond.Wait()
        fmt.Println("cond after")
        cond.L.Unlock()
        wg.Done()
    }()

    go func() {
        time.Sleep(time.Second * 5)
        fmt.Println("释放条件")
        cond.Broadcast()
        wg.Done()
    }()

    wg.Wait()
}
/*
cond wait
释放条件
cond after
*/

sync.Once

函数不管执行多少次，只会被调用一次

package main

import (
    "fmt"
    "sync"
)

func speak() {
    fmt.Println("hello world")
}

func main() {
    var once sync.Once

    for i := 1; i < 5; i++ {
        once.Do(speak) // 只会调用一次
    }
}
/*
hello world
*/

sync.Pool

创建一个对象池

package main

import (
    "fmt"
    "strconv"
    "sync"
)

type proxy struct {
    index int
    addr  string
}

func getAddr(url string) []string {
    urls := make([]string, 0, 100)
    str := "192.168.1."
    for i := 1; i <= 10; i++ {
        urls = append(urls, str+strconv.Itoa(i))
    }
    return urls
}

func main() {
    urls := getAddr("http://www.baidu.com")
    i := 0
    pool := sync.Pool{
        New: func() interface{} {
            i++
            fmt.Println("new : ", i)
            return &proxy{
                i,
                urls[i-1],
            }
        },
    }
    proxy_test := pool.Get() // 获取对象池中的对象，如果没有会new一个
    fmt.Println(proxy_test)
    pool.Put(proxy_test) // 将获取的对象返回创建的对象池

    proxy_test = pool.Get()
    fmt.Println(proxy_test) // 再次获取对象时，会去对象池找，里面存在一个刚刚放回去的就不会再new一个

    proxy_test = pool.Get() // 取出后再次Get,没有找到就会再次new
    fmt.Println(proxy_test)

}
/*
new :  1
&{1 192.168.1.1}
&{1 192.168.1.1}
new :  2        
&{2 192.168.1.2}
*/
