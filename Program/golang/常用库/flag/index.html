flag

命令行解析

绑定参数

XXXTypeVar(&变量, "占位参数即命令行参数", "默认值", "描述信息")这里的XXXType表示数据类型 => int、float、bool、string

var (
    host string
    port string
    h    bool
    help bool
)

flag.BoolVar(&h, "h", false, "帮助信息")
flag.BoolVar(&help, "help", false, "帮助信息")
flag.StringVar(&host, "host", "127.0.0.1", "目标IP")
flag.StringVar(&port, "port", "8080", "目标端口")

返回绑定指针

XXXType("占位参数即命令行参数", "默认值", "描述信息")这里的XXXType表示数据类型 => int、float、bool、string，返回的都是指针数据类型

h := flag.Bool(&h, "h", false, "帮助信息")
help := flag.Bool(&help, "help", false, "帮助信息")
host := flag.String(&host, "host", "127.0.0.1", "目标IP")
port := flag.String(&port, "port", "8080", "目标端口")

/*
*bool : (*bool)(0xc000122058)
*bool : (*bool)(0xc000122059)
*string : (*string)(0xc000104280)
*string : (*string)(0xc000104290)
*/

指定使用说明

flag.Usage = func() {
    fmt.Println("Usage flagTest.exe -hose 127.0.0.1 -p 8080")
    flag.PrintDefaults() // 显示所有已定义的命令行标志的默认设置
}

解析说明

flag.Parse()
flag.Usage()

Usage flagTest.exe -hose 127.0.0.1 -p 8080

  -help
        帮助信息
  -host string
        目标IP (default "127.0.0.1")
  -port string
        目标端口 (default "8080")
*/
