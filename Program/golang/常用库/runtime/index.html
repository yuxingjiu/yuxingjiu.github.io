runtime

runtime包定义了一些协程相关的api

runtime.Gosched()

类似 time.Sleep()

让出CPU时间片，重新等待安排任务（当主协程执行到该函数时，会优先把执行顺序让给字协程执行，字写成执行完毕后，才轮到主协程执行）

package main

import (
    "fmt"
    "runtime"
)

func send(j int) {
    for i := 0; i < 3; i++ {
        fmt.Println(i + j)
    }
}

func main() {
    go send(1)
    for i := 0; i < 3; i++ {
        runtime.Gosched() //有权利执行任务了，让其他字协程去先执行
        fmt.Println("next---")
    }
    fmt.Println("end---")
}

runtime.Goexit()

退出当前协程

package main

import (
    "fmt"
    "runtime"
    "time"
)

func send(j int) {
    for i := 0; i < 3; i++ {
        if i > 1 {
            runtime.Goexit()
        }
        fmt.Println(i + j)
    }
}
func main() {
    go send(1)
    time.Sleep(time.Second)
    fmt.Println("end---")
}

runtime.GOMAXPROCS

默认使用的是全部CPU核

package main

import (
    "fmt"
    "runtime"
    "time"
)

func a() {
    for i := 0; i < 10; i++ {
        fmt.Println(i)
    }
}

func b() {
    for i := 0; i < 10; i++ {
        fmt.Println(i)
    }
}

func main() {
    go a()
    go b()
    time.Sleep(time.Second)
}

package main

import (
    "fmt"
    "runtime"
    "time"

func a() {
    for i := 0; i < 10; i++ {
        fmt.Println("a : ", i)
    }
}

    for i := 0; i < 10; i++ {
        fmt.Println("b : ", i)
    }
}

    runtime.GOMAXPROCS(1)

    go a()

    time.Sleep(time.Second)
