base64

base64编码

  1. 通常说的base64 (0-9a-zA-Z+_)
fmt.Println(base64.StdEncoding.EncodeToString([]byte("成事不说, 遂事不谏, 既往不咎")))
txt, _ := base64.StdEncoding.DecodeString("5oiQ5LqL5LiN6K+0LCDpgYLkuovkuI3osI8sIOaXouW+gOS4jeWSjg==")
fmt.Println(string(txt))
  1. 在URL中 +_ 特殊字符，base64url(+ => -、_ => )
fmt.Println(base64.URLEncoding.EncodeToString([]byte("成事不说, 遂事不谏, 既往不咎")))
txt, _ = base64.URLEncoding.DecodeString("5oiQ5LqL5LiN6K-0LCDpgYLkuovkuI3osI8sIOaXouW-gOS4jeWSjg==")
fmt.Println(string(txt))
  1. 非对齐 3的整数倍 = 补齐
fmt.Println(base64.RawStdEncoding.EncodeToString([]byte("成事不说, 遂事不谏, 既往不咎")))
txt, _ = base64.RawStdEncoding.DecodeString("5oiQ5LqL5LiN6K+0LCDpgYLkuovkuI3osI8sIOaXouW+gOS4jeWSjg")
fmt.Println(string(txt))
fmt.Println(base64.RawURLEncoding.EncodeToString([]byte("成事不说, 遂事不谏, 既往不咎")))
txt, _ = base64.RawURLEncoding.DecodeString("5oiQ5LqL5LiN6K-0LCDpgYLkuovkuI3osI8sIOaXouW-gOS4jeWSjg")
fmt.Println(string(txt))
