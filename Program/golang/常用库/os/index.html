os.OpenFile()

按照指定格式打开文件

格式介绍
O_RDONLY打开只读文件
O_WRONLY打开只写文件
O_RDWR打开既可以读取又可以写入文件
O_APPEND写入文件时将数据追加到文件尾部
O_CREATE如果文件不存在则创建一个新的文件
O_EXCL文件必须不存在，然后会创建一个新的文件
O_SYNC打开同步I/O
O_TRUNC文件打开时可以截断
os.O_WRONLY | os.O_CREATE | O_EXCL       【如果已经存在，则失败】
os.O_WRONLY | os.O_CREATE                【如果已经存在，会覆盖写，不会清空原来的文件，而是从头直接覆盖写】
os.O_WRONLY | os.O_CREATE | os.O_APPEND  【如果已经存在，则在尾部添加写】

package main

import (
    "fmt"
    "os"
)

func main() {
    file, err := os.OpenFile("123.txt", os.O_RDWR|os.O_CREATE, 755)
    if err != nil {
        fmt.Println(err)
    } else {
        fmt.Println(file.Name())
    }
}

os.Stat()

查看文件的属性

package main

import (
    "fmt"
    "os"
)

func main() {

    stat, err := os.Stat("123.txt")
    if err != nil {
        return
    }
    fmt.Println(stat.IsDir())   // 判断是否是文件夹
    fmt.Println(stat.Mode())    // 查看其权限
    fmt.Println(stat.ModTime()) // 最后修改时间
    fmt.Println(stat.Name())    // 查看文件名
    fmt.Println(stat.Size())    // 文件大小
    fmt.Println(stat.Sys())     
}

file.Read()

创建缓存区进行读取

package main

import (
    "fmt"
    "io"
    "os"
)

func main() {

    f, _ := os.OpenFile("1.txt", os.O_RDWR|os.O_CREATE, 755)
    for {
        buff := make([]byte, 3)
        n, err := f.Read(buff)
        fmt.Println("n : ", n)
        fmt.Println("value : ", string(buff))
        if err == io.EOF {
            break
        }
    }
    _ = f.Close()

}

进程代码

package main

import (
    "fmt"
    "os"
    "time"
)

func main() {
    // 设置进程
    cmd := &os.ProcAttr{
        // Files指定新进程继承的活动文件对象
        Files: []*os.File{ // 前三个分别为标准输入、标准输出、标准错误输出
            os.Stdin,
            os.Stdout,
            os.Stderr,
        },
        Env: os.Environ(), // 新进程的环境变量
    }
    // 开始一个新进程
    process, err := os.StartProcess("D:\\Typora\\Typora.exe", []string{"D:\\Typora\\Typora.exe", ""}, cmd)
    if err != nil {
        fmt.Println(err)
    }
    fmt.Println(process)
    fmt.Println("进程ID : ", process.Pid) 
    findProcess, _ := os.FindProcess(process.Pid) // 根据进程ID找进程
    fmt.Println(findProcess)

    time.AfterFunc(time.Second*20, func() { // 等待20秒，执行函数
        _ = process.Signal(os.Kill) // 向进程发送指令，杀掉进程
    })
    wait, _ := process.Wait() // 等待进程结束，返回进程状态
    fmt.Println(wait)
}

os.Clearenv()

清空环境变量

package main

import (
    "fmt"
    "os"
)

func main() {
    _ = os.Setenv("env1", "env1")
    _ = os.Setenv("env2", "/usr/env")
    fmt.Println("没有请空前 : ")
    _, b := os.LookupEnv("env1")
    fmt.Println("env1 : ", b)
    _, b2 := os.LookupEnv("env2")
    fmt.Println("env2 : ", b2)
    os.Clearenv()
    fmt.Println("清空后 : ")
    _, b = os.LookupEnv("env1")
    fmt.Println("env1 : ", b)
    _, b2 = os.LookupEnv("env2")
    fmt.Println("env2 : ", b2)
}

没有请空前 :
env1 :  true
env2 :  true
清空后 :
env1 :  false
env2 :  false
