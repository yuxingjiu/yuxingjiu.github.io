创建错误

创建错误类型变量

fmt.Errorf("错误的原因")

通过errors包创建

errors.New("自定义错误")

package main

import (
    "errors"
    "fmt"
)

func js(n1, n2 int) (int, error) {
    if n2 == 0 {
        return 0, errors.New("除数不能为0")
    } else {
        return n1 / n2, nil
    }
}

func main() {
    if i, err := js(1, 0); err == nil {
        fmt.Println(i)
    } else {
        fmt.Println(err)
    }
}
// 除数不能为0

触发异常

panic("错误原因")

解决异常

recover必须是在延迟执行函数内

package main

import (
    "fmt"
)

func test() (err error) {
    defer func() {
        if panicError := recover(); panicError != nil {
            err = fmt.Errorf("%s", panicError)
        }
    }()
    fmt.Println("test start~~~")
    panic("自定义错误")
    fmt.Println("test end~~~")
    return
}

func main() {
    fmt.Println("main start")
    fmt.Printf("%T, %#v\n", test(), test())
    fmt.Println("main end")
}

main start
test start~~~
test start~~~
*errors.errorString, &errors.errorString{s:"自定义错误"}
main end

这样当存在异常时，可以使用recover()函数捕获，当不存在异常时，为<nil>
