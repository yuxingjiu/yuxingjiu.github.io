占位符

类型描述
%b输出二进制数
%d输出十进制数
%o输出八进制数
%x/%X输出十六进制数
%T查看数据类型
%s查看字符串
%U查看rune类型
%c查看字符
%v打印变量的值，不管什么数据类型
%#v打印变量的值，以及变量的特征符号【字符串的引号等】
%p打印指针
package main

import "fmt"

// fmt的占位符

func main() {
    var n = 123
    fmt.Printf("%v\n", n)
    fmt.Printf("%T\n", n)
    fmt.Printf("%b\n", n)
    fmt.Printf("%o\n", n)
    fmt.Printf("%d\n", n)
    fmt.Printf("%x\n", n)
    var m = "hello world"
    fmt.Printf("%s\n", m)
    fmt.Printf("%v\n", m)
}
