流程控制

流程控制是每种编程语言控制逻辑走向和执行次序的重要部分，流程控制可以说是一门语言的经脉

Go语言中最常见的流程控制有if和for，而switch和goto主要是为了简化代码降低重复代码而生的结构，属于扩展类的流程控制

关键字

break

continue

goto

if else

基本写法

age := 10
if age < 18{
    分支1
}else if age > 18{
    分支2
}else{
    分支3
}
fmt.Println(age) // 可以执行

特殊写法

if age := 10; age < 18{
    分支1
}else if age > 18{
    分支2
}else{
    分支3
}
fmt.Println(age) // 无法执行，因为age定义在了if的作用域内

for循环

Go语言中的所有循环类型皆可以使用for关键字来完成

基本格式

for 初始条件; 条件表达式; 结束语句 {
    循环体语句
}

条件较大时返回true时循环体不停的进行循环，Hi到条件表达式返回false自动退出循环

for i := 0; i < 10; i++ {
    fmt.Println(i)
}


i := 0
for i < 10; i++ {
    fmt.Println(i)
}

i := 0
for i < 10 {
    fmt.Println(i)
    i++
}

f​or循环的初始语句可以被忽略，但是初始语句后的分号必须要写

for range

键值循环

Go语言中可以使用for range遍历数组、切片、字符串、map及通道(channel)

通过for range遍历的返回值有以下规律

  1. 数组、切片、字符串返回索引和值
  2. map返回键和值
  3. 通道(channel)只返回通道内的值
for i,v := range value {
    fmt.Println(i,":", v)
}

switch

基本写法

package main

import "fmt"

// switch

func main() {
    age := 18
    switch age {
        case 17:
        fmt.Println("未成年")
        case 18:
        fmt.Println("成年")
        case 19:
        fmt.Println("长大了")
    }
}
// 会直接输出成年

多种条件

package main

import "fmt"

func main() {
    str := ""
    _, _ = fmt.Scanln(&str)
    switch str {
    case "yes", "y", "Y":
        fmt.Println("正确")
    case "no", "n", "N":
        fmt.Println("错误")
    }
}
// 此时只要输入3个选项中的一个就可以进入case

分支表达语句

package main

import "fmt"

// switch

func main() {
    age := 18
    switch {
        case age < 6:
        fmt.Println("幼儿园")
        case age >= 6 && age < 12:
        fmt.Println("小学生")
        case age >= 12 && age < 15:
        fmt.Println("初中生")
        case age >= 15 && age < 18:
        fmt.Println("高中生")
        case age >= 18 && age < 22:
        fmt.Println("大学生")
        default:
        fmt.Println("~~~~")
    }
}

实现字节转换

package main

import (
    "fmt"
    "os"
)

func FloderDemo01() {
    // files, err := ioutil.ReadDir(".")
    files, err := os.ReadDir(".")
    if err != nil {
        fmt.Println(err)
    }
    for _, f := range files {
        if f.IsDir() {
            fmt.Println("文件夹 : ", f.Name())
        } else {
            fmt.Println("文件 : ", f.Name())
        }

    }
}

const (
    _   = iota
    KB  = 1 << (10 * iota)
    MB  = 1 << (10 * iota)
    GB  = 1 << (10 * iota)
    TB  = 1 << (10 * iota)
    PB  = 1 << (10 * iota)
    bit = 8
)

func FormatSize(sizes int64) {
    size := float64(sizes)

    fmt.Println(KB)
    fmt.Println(MB)
    fmt.Println(TB)
    fmt.Println(PB)

    switch {
    case size > 0 && size < bit:
        fmt.Printf("%v => %.3v bit\n", size, size)
    case size >= bit && size < KB:
        fmt.Printf("%v => %.3v kb\n", size, size/bit)
    case size >= KB && size < MB:
        fmt.Printf("%v => %.3v mb\n", size, size/KB)
    case size >= MB && size < GB:
        fmt.Printf("%v => %.3v gb\n", size, size/MB)
    case size >= GB && size < TB:
        fmt.Printf("%v => %.3v tb\n", size, size/GB)
    case size >= TB && size < PB:
        fmt.Printf("%v => %.3v pb\n", size, size/TB)
    default:
        fmt.Printf("null\n")
    }
}


func main() {
    FormatSize(1024 * 1024 * 1024 * 20) // 20tb
}

goto

跳转到指定标签

goto语句通过标签进行代码间的无条件跳转，goto语句可以在快速调处循环、避免重复退出上有一定的帮助

Go语言中使用的got语句能简化一些代码的实现过程，如双层嵌套的for循环要退出时

package main

import "fmt"

// goto

func main() {
    // 这个break只能退出内层的循环，不是整体退出
    for i := 1; i < 3; i++ {
        for j := 'A'; j <= 'Z'; j++ {
            if j == 'C' {
                break
            }
            fmt.Printf("%d - %c\n", i, j)
        }
    }
    for i := 1; i < 3; i++ {
        for j := 'A'; j <= 'Z'; j++ {
            if j == 'C' {
                goto lab
            }
            fmt.Printf("%d - %c\n", i, j)
        }
    }
    fmt.Println("123") // 123就不会被输出
    lab: // lable标签，执行到goto lab就会跳到lab标签处，继续执行以下的代码
    fmt.Println("over")
    fmt.Println("456")
}

死循环

package main

import "fmt"

// goto

func main() {
    for i := 1; i < 3; i++ {
        for j := 'A'; j <= 'Z'; j++ {
            if j == 'C' {
                goto lab1
            }
            fmt.Printf("%d - %c\n", i, j)
        }
    }
    fmt.Println("123")
    lab1:
    fmt.Println("over")
    lab2:
    fmt.Println("456")
    for i := 1; i < 3; i++ {
        for j := 'A'; j <= 'Z'; j++ {
            if j == 'C' {
                goto lab2
            }
            fmt.Printf("%d - %c\n", i, j)
        }
    }
}

break

break只能跳出当前循环，如果存在多层循环时，可以仿造goto的写法跳出多层循环

package main

import "fmt"

func main() {
BREAK:
    for i := 0; i < 2; i++ {
        fmt.Println(i, "---------------------------")
        for j := 0; j < 3; j++ {
            fmt.Println(j)
            if j == 2 {
                break BREAK
            }
        }
    }
}

/*
0 ---------------------------
0
1
2
*/

通过break后面加参数，可以达到跳出多层循环

continue

continue时跳过本次循环，如果不想执行本层循环，但是又不想直接跳出最外面（即实现break的功能）

package main

import "fmt"

func main() {
CONTINUE:
    for i := 0; i < 2; i++ {
        fmt.Println(i, "---------------------------")
        for j := 0; j < 3; j++ {
            fmt.Println(j)
            if j == 2 {
                continue CONTINUE
            }
        }
    }
}
/*
0 ---------------------------
0
1
2
1 ---------------------------
0
1
2
*/
