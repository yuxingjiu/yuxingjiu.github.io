HTTP请求

http 是一种无状态协议，服务器不会维护每个请求的状态，而是通过多种方式分钟其状态，这些方式可能包括标识符、cookie、HTTP 标头等，客户端和服务器有责任协商和验证状态

import (
    "bufio"
    "fmt"
    "io"
    "log"

)

结构化解析

内容解析

package httpclient

import (
    "bufio"
    "fmt"
    "io"
    "log"
    "net/http"
)

func clientGet() {
    r, err := http.Get("https://www.baidu.com/robots.txt")
    if err != nil {
        log.Fatalln("请求错误")
    }
    body := bufio.NewReader(r.Body)
    b, _ := io.ReadAll(body) // 解析
    fmt.Printf("b: %v\n", string(b))
    r.Body.Close()
}

通过 GET 请求返回一个变量 r(*Response 结构体类型)的响应后，r 类型包含一个可输出的参数 Body （io.ReadCloser），io.ReadCloser 充当 io.Reader 和 io.Closer 的接口，或者是需要实现 Close()函数以及关闭 reader 并执行任何清理的接口，从 io.ReadCloser()读取数据后，需要在响应正文上嗲用 Close()函数关闭（也可以添加 defer 进行关闭）

json 解析

定义一个 Value 的结构体，其中包含了服务器响应中的预期元素（这里只写了部分），主函数宪法起请求，解码响应正文后，通过访问可输出的数据类型 Value 来查询结构体 Value

也就是先定义一个用来表示响应数据的结构体，然后把数据解码到该结构体中

package httpclient

import (
    "encoding/json"
    "log"
    "net/http"
)

type Value struct {
    Page    int        `json:"page"`
    Results [][]string `json:"results"`
}

func Client() {
    https := "https://fofa.info/api/v1/search/all?email=fofa_email&key=fofa_key&qbase64=dGl0bGU9ImJpbmci"
    var (
        client http.Client
        value  Value
    )
    r, err := http.NewRequest("GET", https, nil)
    if err != nil {
        log.Fatalln("错误数据")
    }
    r.Header.Add("Content-Type", "application/json")
    resp, err := client.Do(r)
    if err != nil {
        log.Fatalln("请求错误")
    }
    err = json.NewDecoder(resp.Body).Decode(&value)
    if err != nil {
        log.Fatalln("解析错误")
    }
    log.Printf("page -> %d\n", value.Page)
    log.Printf("result -> %s\n", value.Results)

    defer resp.Body.Close()
}

msf.go

package msfrpc

import (
    "bytes"
    "fmt"
    "log"
    "net/http"

    "gopkg.in/vmihailenco/msgpack.v2"
)

// 请求结构体
type SessionListReq struct {
    _msgpack struct{} `msgpack:",asArray"`
    Method   string
    Token    string
}

// 响应结构体
type SessionListRes struct {
    ID          uint32 `msgpack:",omitempty"`
    Type        string `msgpack:"type"`
    TunnelPeer  string `msgpack:"tunnel_peer"`
    ViaExploit  string `msgpack:"via_exploit"`
    ViaPayload  string `msgpack:"via_payload"`
    Description string `msgpack:"desc"`
    Info        string `msgpack:"info"`
    Workspace   string `msgpack:"workspace"`
    SessionHost string `msgpack:"session_host"`
    SessionPort int    `msgpack:"session_port"`
    Username    string `msgpack:"username"`
    UUID        string `msgpack:"uuid"`
    ExploitUUID string `msgpack:"exploit_uuid"`
}

// 登录Metsploit
type loginReq struct {
    _msgpack struct{} `msgpack:",asArray"`
    Method   string
    Username string
    Password string
}

type loginRes struct {
    Result       string `msgpack:"result"`
    Token        string `msgpack:"token"`
    Error        bool   `msgpack:"error"`
    ErrorClass   string `msgpack:"error_class"`
    ErrorMessage string `msgpack:"error_message"`
}

// 登出Metsploit
type logoutReq struct {
    _msgpack    struct{} `msgpack:",asArray"`
    Method      string
    Token       string
    LogoutToken string
}

type logoutRes struct {
    Result string `msgpack:"result"`
}

type Metasploit struct {
    host  string
    user  string
    pass  string
    token string
}

func New(host, user, pass string) (*Metasploit, error) {
    msf := &Metasploit{
        host: host,
        user: user,
        pass: pass,
    }
    err := msf.Login()
    if err != nil {
        return nil, err
    }
    return msf, nil
}

func (msf *Metasploit) send(req interface{}, res interface{}) error {
    buf := new(bytes.Buffer)
    msgpack.NewEncoder(buf).Encode(req)
    s := fmt.Sprintf("http://%s/api", msf.host)
    r, err := http.Post(s, "binary/message-pack", buf)
    if err != nil {
        log.Fatalln("请求错误")
    }
    defer r.Body.Close()
    err = msgpack.NewDecoder(r.Body).Decode(&res)
    if err != nil {
        log.Fatalln("编码错误")
    }
    return nil
}

func (msf *Metasploit) Login() error {
    ctx := &loginReq{
        Method:   "auth.login",
        Username: msf.user,
        Password: msf.pass,
    }
    var res loginRes
    err := msf.send(ctx, &res)
    if err != nil {
        log.Fatalln("请求数据失败")
    }
    msf.token = res.Token
    return nil
}
func (msf *Metasploit) Logout() error {
    ctx := &logoutReq{
        Method:      "auth.logout",
        Token:       msf.token,
        LogoutToken: msf.token,
    }
    var res logoutRes
    err := msf.send(ctx, &res)
    if err != nil {
        log.Fatalln("请求数据失败")
    }
    msf.token = ""
    return nil
}

func (msf *Metasploit) SessionList() (map[uint32]SessionListRes, error) {
    req := &SessionListReq{
        Method: "session.list",
        Token:  msf.token,
    }
    res := make(map[uint32]SessionListRes)
    err := msf.send(req, &res)
    if err != nil {
        log.Fatalln("请求数据失败")
    }
    for id, session := range res {
        session.ID = id
        res[id] = session
    }
    return res, nil
}

服务器

http.HandleFunc()

http.HandleFunc(pattern string, handler func(ResponseWriter, *Request))

该函数接收两个参数，一个字符串（该字符串是我们指示服务器去寻找的URL路径模式）和一个实际处理请求的函数

该函数既可以为匿名内联函数

还可以是自定义函数

而这个函数（自定义或内联）需要接收两个参数

  1. 第一个是http.ResponWriter （用于写入对请求的响应）
  2. 第二个是指向http.Request 的指针 （它将允许我们从传入的请求中读取信息）

不是在主函数中调用manage，而指示告诉HTTP服务器对 /hello 的所有请求都是由manage()函数处理

http.ListenAndServer()

http.ListenAndServe(addr string, handler Handler) error

该函数接收两个参数，一个是字符串第二个是http.Handler

通过使用第一个参数作为地址来启动HTTP服务器（这里用的8080）

package httpserver

import (
    "fmt"
    "net/http"
)

func manage(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "hello %s\n", r.URL.Query().Get("name"))
}

func HttpServer() {
    http.HandleFunc("/hello", manage)
    http.ListenAndServe(":8080", nil)
}

路由器

定义一个没有任何字段的名为Router的结构体，用来实现http.Handler，想要实现http.Handler得先定义方法ServerHTTP()

package httpserver

import (
    "fmt"
    "net/http"
)

type Router struct{}

func (router *Router) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    switch r.URL.Path {
    case "/a":
        fmt.Fprintf(w, "这里是A界面")
    case "/b":
        fmt.Fprintf(w, "这里是B界面")
    case "/c":
        fmt.Fprintf(w, "这里是C界面")
    default:
        fmt.Fprintf(w, "404")
    }
}

func HttpRouter() {
    var router Router
    http.ListenAndServe(":8080", &router)
}

请求路径匹配传递参数

打印/mpy/路径后面的参数

在定义请求路径时，可以使用花括号来定义请求参数，可以将此看为占位符，之后使用 mux.Vars()将请求对象传递给它，此时会返回 map[string]string 类型，是请求参数名称到他们各自值的映射

package httpserver

import (
    "fmt"
    "net/http"

    "github.com/gorilla/mux"
)

func MuxRouterPath() {
    r := mux.NewRouter()
    r.HandleFunc("/mpy/{user}", func(w http.ResponseWriter, r *http.Request) {
        user := mux.Vars(r)["user"]
        fmt.Fprintf(w, "hello %s", user)
    })
    http.ListenAndServe(":8080", r)
}

中间件

中间件是一种包装程序，它将在所有传入请求上执行，而与目标函数无关

package httpserver

import (
    "fmt"
    "log"
    "net/http"
)

type logger struct {
    Inner http.Handler
}

func (l *logger) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    log.Println("start")
    l.Inner.ServeHTTP(w, r)
    log.Println("end")
}

func hello(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "hello")
}
func HttpMiddleWare() {
    f := http.HandlerFunc(hello)
    l := logger{Inner: f}
    http.ListenAndServe(":8080", &l)
}

每次服务器收到请求都会在服务器上记录一些信息并调用函数hello()

negroni中间件

package httpserver

import (
    "net/http"

    "github.com/gorilla/mux"
    "github.com/urfave/negroni"
)

func NegroniRouter() {
    r := mux.NewRouter()
    n := negroni.Classic()
    n.UseHandler(r)
    http.ListenAndServe(":8080", n)
}
  1. 先创建路由器
  2. 创建negroni.Classic()使用默认的中间件（包括请求记录器、紧急情况下的进行拦截和恢复的中间件，以及服务于同一目录中公用文件夹的中间件，而negroni.New()则不会创建任何默认的中间件）
  3. 通过n.UseHandler()将路由器添加到中间件堆栈

身份认证

package httpserver

import (
    "context"
    "fmt"
    "net/http"

    "github.com/gorilla/mux"
    "github.com/urfave/negroni"
)

type userInfo struct {
    Username string
    Password string
}

func (info *userInfo) ServeHTTP(w http.ResponseWriter, r *http.Request, next http.HandlerFunc) {
    userinfo := make(map[string]string, 10)
    user := r.URL.Query().Get("user")
    pass := r.URL.Query().Get("pass")
    userinfo[user] = pass
    if user != info.Username || pass != info.Password {
        http.Error(w, "user or pass error", 404)
        return
    }
    ctx := context.WithValue(r.Context(), "xx", userinfo)
    r = r.WithContext(ctx)
    next(w, r)
}

func manages(w http.ResponseWriter, r *http.Request) {
    info := r.Context().Value("xx").(map[string]string)
    for user, pass := range info {
        fmt.Fprintf(w, "Hi %v\nPass %v\n", user, pass)
    }

}

func LoginNegroniRouter() {
    r := mux.NewRouter()
    r.HandleFunc("/hello", manages).Methods("Get")
    n := negroni.Classic()
    n.Use(&userInfo{
        Username: "admin",
        Password: "pass",
    })
    n.UseHandler(r)
    http.ListenAndServe(":8080", n)
}
