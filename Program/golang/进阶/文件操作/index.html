package main

import (
    "fmt"
    "io"
    "os"
)

func main() {
    text, err := os.Open("H:\\GoObject\\demo01\\day05\\1.txt")
    if err != nil {
        return
    }
    defer text.Close()
    context := make([]byte, 1024)
    for {
        c, err := text.Read(context)
        if err == io.EOF {
            break
        }
        fmt.Println(string(context[:c]), len(context[:c]))
    }
}
  1. os.OpenFile() : 按照指定格式打开文件
格式介绍
O_RDONLY打开只读文件
O_WRONLY打开只写文件
O_RDWR打开既可以读取又可以写入文件,，打开文件从前到后的顺序写入，会覆盖
O_APPEND写入文件时将数据追加到文件尾部
O_CREATE如果文件不存在则创建一个新的文件
O_EXCL文件必须不存在，然后会创建一个新的文件
O_SYNC打开同步I/O
O_TRUNC文件打开时可以截断
os.O_WRONLY | os.O_CREATE | O_EXCL       【如果已经存在，则失败】
os.O_WRONLY | os.O_CREATE                【如果已经存在，会覆盖写，不会清空原来的文件，而是从头直接覆盖写】
os.O_WRONLY | os.O_CREATE | os.O_APPEND  【如果已经存在，则在尾部添加写】

package main

import (
    "fmt"
    "os"
)

func main() {
    file, err := os.OpenFile("123.txt", os.O_RDWR|os.O_CREATE, 755)
    if err != nil {
        fmt.Println(err)
    } else {
        fmt.Println(file.Name())
    }
}

查看目录下的文件

file.Readdirnames() 模拟 file.Walk

package main

import (
    "fmt"
    "io"
    "os"
    "strings"
)

type FileFilter func(path string) bool
type FilePrint func(path string)

func readFile(path string) string {
    file, err := os.Open(path)
    if err != nil {
        return ""
    }
    defer file.Close()
    txt := make([]byte, 0, 1024*1024)
    cxt := make([]byte, 1024)
    for {
        n, err := file.Read(cxt)
        if err == io.EOF {
            break
        }
        txt = append(txt, cxt[:n]...)
    }
    return string(txt)
}

func Dir(path string, filter FileFilter, printer FilePrint) {
    file, err := os.Open(path) // 打开一个文件
    if err != nil {
        return
    }
    defer file.Close()                     // 延迟关闭文件
    filename, err := file.Readdirnames(-1) // 获取path路径下的所有文件名(-1表示全部)
    if err != nil {
        return
    }
    for _, v := range filename { // 遍历每个文件，并判断是否是目录
        filepath := path + "/" + v
        stat, err := os.Stat(filepath)
        if err != nil {
            return
        }
        if stat.IsDir() { // 判断文件是否是目录
            Dir(filepath, filter, printer)
        } else {
            if filter == nil || filter(filepath) { // 过滤器
                if printer != nil { // 输出文件信息
                    printer(filepath)
                }
            }
        }
    }
}

func main() {
    Dir("H:\\GoObject\\demo01\\modules", func(path string) bool {
        return strings.ContainsAny(path, "hi")
    }, func(path string) {
        fmt.Println("file : ", path)
        fmt.Println(readFile(path))
    })

}

方法一

利用for循环读取文件存储到切片中

package demo1

import (
    "fmt"
    "io"
    "os"
)

func main() {
    path := "D:\\笔记\\code\\1.txt"
    
    file, err := os.Open(path)
    if err != nil {
        fmt.Println(err)
    }
    defer file.Close()

    ctx := make([]byte, 0, 1024*1024)
    buffer := make([]byte, 100)

    for {
        n, err := file.Read(buffer)
        if err == io.EOF {
            break
        }
        ctx = append(ctx, buffer[:n]...)
    }
    fmt.Println(string(ctx))
}

数据编码存储

主程序

package main

import (
    "bufio"
    "encoding/csv"
    "encoding/gob"
    "fmt"
    "io"
    "os"
    "strconv"
    "time"
)

type Task struct {
    ID        int
    Name      string
    Desc      string
    Status    bool
    StartTime string
    EndTime   string
}

const (
    TimeLayout = "2006/01/02 15:04:05"
)

var (
    tasks = make([]*Task, 0, 10)
)

func CreateTask(name string) {

    task := Task{
        ID:        len(tasks) + 1,
        Name:      name,
        Status:    false,
        StartTime: time.Now().Format(TimeLayout),
        EndTime:   "",
    }
    tasks = append(tasks, &task)
}

func ChangeTask(s bool, task *Task) {
    if s {
        task.Status = true
        task.EndTime = time.Now().Format(TimeLayout)
    }
}

func main() {

    // 创建任务

    CreateTask("打游戏")
    CreateTask("学习")

    // 修改任务状态

    ChangeTask(true, tasks[1])

    // gob写入数据到文件和从文件中获取数据

    GobDecoder()

    GobDecoder()

    // csv写入数据到文件和从文件中获取数据

    CSVDecoder()
    
    CSVEncoder()

}

package main

import (
    "bufio"
    "encoding/csv"
    "fmt"
    "io"
    "os"
    "strconv"
)

func CSVEncoder() {
    f, err := os.OpenFile("D:\\笔记\\code\\tmp\\1.csv", os.O_RDWR|os.O_CREATE|os.O_APPEND, 0755)
    if err != nil {
        fmt.Println(err)
    }
    defer f.Close()

    w := bufio.NewWriter(f)

    defer w.Flush()

    csvWriter := csv.NewWriter(w)
    csvWriter.Write([]string{"ID", "任务名", "状态", "开始时间", "结束时间"})
    for _, task := range tasks {
        csvWriter.Write([]string{
            strconv.Itoa(task.ID),
            task.Name,
            strconv.FormatBool(task.Status),
            task.StartTime,
            task.EndTime,
        })
    }
}
