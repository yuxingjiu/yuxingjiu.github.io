builder

strings.Builder

string.Builder : 定义strings.Builder结构体对象，类似在内存中的流对象（写文件操作）

写操作

builder.Write()、builder.WriteString()、builder.WriteByte()、builder.WriteRune()

package main

import (
    "fmt"
    "strings"
)

func main() {
    var builder strings.Builder
    builder.Write([]byte("写文件\n"))
    builder.Write([]byte("操作\n"))
    builder.WriteString("写文件操作")
}

将写入的内容转为字符串

builder.String()

fmt.Println(builder.String())

操作
写文件操作
*/

bytes.Buffer

构建buffer结构体指针对象

bytes.NewBuffer()和bytes.NewBufferString() : 前者通过字符切片构建buffer结构体指针对象，后者是通过字符串构建buffer结构体指针对象

package main

import (
    "bytes"
    "fmt"
)

func main() {
    // 构建buffer结构体指针对象
    byteBuffer := bytes.NewBuffer([]byte("bytes : "))
    strBuffer := bytes.NewBufferString("bytes : ")
}

写操作

buffer.Write()

byteBuffer.Write([]byte(" => bytes.NewBuffer()"))
strBuffer.Write([]byte(" => bytes.NewBufferString()"))

读操作

buffer.Read()

ctx := make([]byte, 100)
n, err := byteBuffer.Read(ctx)
fmt.Println(n, err, string(ctx[:n])) // 29 <nil> bytes :  => bytes.NewBuffer()
n, err = strBuffer.Read(ctx)
fmt.Println(n, err, string(ctx[:n])) // 35 <nil> bytes :  => bytes.NewBufferString()

将指针返回字符串开始位置

reader.Seek(0, io.SeekStart)

package main

import (
    "fmt"
    "io"
    "strings"
)

func main() {
    reader := strings.NewReader("123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ")
    ctx := make([]byte, 8)
    for {
        n, err := reader.Read(ctx)
        if err != nil {
            break
        }
        fmt.Println(n, err, string(ctx[:n]))
    }
    fmt.Println("===============================")
    _, _ = reader.Seek(0, io.SeekStart)
    n, err := reader.Read(ctx)
    fmt.Println(n, err, string(ctx[:n]))
}
/*
8 <nil> 12345678
8 <nil> 9abcdefg
8 <nil> hijklmno
8 <nil> pqrstuvw
8 <nil> xyzABCDE
8 <nil> FGHIJKLM
8 <nil> NOPQRSTU
5 <nil> VWXYZ
===============================
8 <nil> 12345678
*/

重新设置字符串

reader.Reset() : 重新设置字符串，之后读取会重新读取

reader := strings.NewReader("123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ")
reader.Reset("abcdefghijklmnopqrstuvwxyz")
n, err := reader.Read(ctx)
fmt.Println(n, err, string(ctx[:n])) // 8 <nil> abcdefgh

读取内容到指定位置

reader.WriteTo() : 读取内容到指定位置，可以模拟实现print的效果

reader.WriteTo(os.State)

模拟fmt.Print()

package main

import (
    "bytes"
    "fmt"
    "io"
    "os"
)

func main() {
    f, err := os.OpenFile("D:\\笔记\\code\\1.txt", os.O_RDWR, 0755)
    ctx := make([]byte, 10)
    if err != nil {
        fmt.Println(err)
    }
    defer f.Close()
    for {
        n, err := f.Read(ctx)
        if err == io.EOF {
            break
        }
        reader := bytes.NewReader(ctx[:n])
        reader.WriteTo(os.Stdout)
    }
}

io.Copy()

将一个输入流复制到另一个输出流中

_, err = io.Copy(file_dst, file_src)

会返回一个写入的字节数量和err，前者可以忽略，查看后者是否写入成功

实现复制文件

package main

import (
    "bytes"
    "fmt"
    "io"
    "os"
    "path/filepath"
)

func dir_file(path, f string) bool {
    file, err := os.Open(path)
    if err != nil {
        fmt.Println(err)
    }
    defer file.Close()
    fs, _ := file.Readdirnames(-1)
    for _, v := range fs {
        if v == f {
            return false
        }
    }
    return true
}

func main(srcpath, dstpath string) {
    file_src, err := os.Open(srcpath)
    if err != nil {
        fmt.Println(err)
        return
    }
    defer file_src.Close()

    _, f_s := filepath.Split(srcpath)
    d_d, f_d := filepath.Split(dstpath)

    if f_d == "" {
        dstpath = filepath.Join(dstpath, f_s)
    }

    b := dir_file(d_d, f_d)

    if !b {
        fmt.Println("目标文件存在")
        return
    }

    file_dst, _ := os.Create(dstpath)
    defer file_dst.Close()
    ctx := make([]byte, 50)
    for {
        n, err := file_src.Read(ctx)
        if err == io.EOF {
            break
        }
        file_dst.Write(ctx[:n])
    }
}

io.CopyBuffer()

io.Copy()无法控制缓冲区，而io.CopyBuffer()可以自定义一个Buffer

buffer := make([]byte, 1024*1024)
_, err = io.Copy(file_dst, file_src, buffer)

io.CopyN()

复制前N个字符

n := make([]byte, 1024)
_, err = io.Copy(file_dst, file_src, buffer) // 从src复制n个字符到dst

io.MultiWriter()

将一个输出输出到多个输出流中

package main

import (
    "io"
    "os"
)

func main() {
    f1, _ := os.Create("tmp/1.log")
    f2, _ := os.Create("tmp/2.log")

    defer f1.Close()
    defer f2.Close()

    writer := io.MultiWriter(f1, f2)
    
    writer.Write([]byte("这是一些测试文件"))
}

io.MultiReader()

将多个输出流输出到一个输出流中

package main

import (
    "io"
    "os"
)

func main() {
    f1, _ := os.Open("tmp/1.log")
    f2, _ := os.Open("tmp/2.log")

    defer f1.Close()
    defer f2.Close()

    reader := io.MultiReader(f1, f2)
    ctx := make([]byte, 100)

    for {
        n, err := reader.Read(ctx)
        if err == io.EOF {
            break
        }
        fmt.Print(string(ctx[:n])) // 先读取第一个流，也就是第一个文件1.log，然后在读第二个文件2.log
    }
    fmt.Println()

}

package main

    "bufio"

    "os"
)

func main() {

    if err != nil {
        fmt.Println(err)
    }
    defer file.Close()

    scan := bufio.NewScanner(file)
    for scan.Scan() { // 每次读取一行，换行停止
        fmt.Println(scan.Text()) // 输出读取的内容
    }
}

io.Reader&io.Writer

无论使用的是TCP、HTTP、文件系统还是任何其他方式，都需要使用两种类型 io.Reader 和 io.Writer ，对于所有输入/输出（I/O）任务来说，这两种数据类型都是必不可少的，它们都是Go内置io包的一部分，是所有本地或网络数据传输的基础

type Reader interface {
    Read(p []byte) (n int, err error)
}
type Writer interface {
    Write(p []byte) (n int, err error)
}

这两种类型都被定义为接口类型，意味着不能直接被实例化，每种数据类型包含单个导出函数的定义，分别是Read、Write，将这种函数理解为抽象方法，该抽象方法必须在某种数据类型上得到实现才能被视为Reader或Writer

type FooReader struct{}
func (fooReader *FooReader) Read(p []byte) (int, error){
    return len(dataReadFromSomewhere), nil
}
type FooWriter struct{}
func (fooWriter *FooWriter) Read(p []byte) (int, error){
    return len(dataReadFromSomewhere), nil
}

实现io.Reader&io.Reader

package tcpproxy

import (
    "fmt"
    "os"
)

type FooReader struct{}
type FooWriter struct{}

func (fooReader *FooReader) Read(b []byte) (int, error) {
    fmt.Print("in >")
    return os.Stdin.Read(b)

}

func (fooWriter *FooWriter) Write(b []byte) (int, error) {
    fmt.Print("out>")
    return os.Stdout.Write(b)
}

func RW() {
    var (
        reader FooReader
        writer FooWriter
    )
    input := make([]byte, 4096)
    readLen, err := reader.Read(input)
    if err != nil {
        fmt.Println(err)
    }
    fmt.Printf("Read %d bytes from stdin\n", readLen)

    writeLen, err := writer.Write(input)
    if err != nil {
        fmt.Println(err)
    }
    fmt.Printf("Write %d byted to stdout\n", writeLen)
}

该程序定义了两种自定义类型 ：FooReader、FooWriter，在每种类型上，为FooReader定义了函数Read(b []byte)，为FooWriter定义了Write(b []byte)，在这里两个函数都从stdin读取并写入stdout

in >hello world

Read 13 bytes from stdin

out>hello world

Write 4096 byted to stdout

将数据才从Reader复制到Writer是一种非常常见的模式，促使Go的io包中提供了依噶函数Copy

io.Copy(dst io.Writer, src io.Reader) (writer int64, error)
// dst表示目的，src表示源
package tcpproxy

import (
    "fmt"
    "io"
)

type FooReader struct{}
type FooWriter struct{}

func (fooReader *FooReader) Read(b []byte) (int, error) {
    fmt.Print("in >")
    return os.Stdin.Read(b)

}

func (fooWriter *FooWriter) Write(b []byte) (int, error) {
    fmt.Print("out>")
    return os.Stdout.Write(b)
}

func CopyRW() {
    var (
        reader FooReader
        writer FooWriter
    )
    if _, err := io.Copy(&writer, &reader); err == nil {
        fmt.Println(err)
    }
}
