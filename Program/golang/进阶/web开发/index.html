应用开发

http包提供了HTTP服务器和客户端的开发接口，内置web服务器

服务端

针对web服务端开发流程为

  1. 处理函数 func(http.ResponseWriter, *http.Request){}
  2. 绑定URL关系 http.HandleFunc()、http.Handle()
  3. 启动web服务器 http.ListenAndServe()

绑定URL关系可以通过很多函数设置处理器去处理

处理器创建

http.HandleFunc()

  • 处理器函数
http.HandleFunc(pattern string, handler func(http.ResponseWriter, *http.Request))

该函数接收两个参数，一个字符串（该字符串是我们指示服务器去寻找的URL路径模式）和一个实际处理请求的函数

而这个函数（自定义或内联）需要接收两个参数

  1. 第一个是http.ResponWriter [ 用于写入对请求的响应 ]
  2. 第二个是指向http.Request 的指针 [ 它将允许我们从传入的请求中读取信息 ]

不是在主函数中调用manage，而指示告诉HTTP服务器对 /hello 的所有请求都是由manage()函数处理

该函数既可以为匿名内联函数

还可以是自定义函数

package main

import (
    "fmt"
    "net/http"
    "os"
)

func main() {

    http.HandleFunc("/", func(writer http.ResponseWriter, request *http.Request) { // 绑定匿名函数和 "/" 的关系
        file, err := os.ReadFile("H:\\Object\\GoObject\\demo\\demo11\\httpv1\\index.html") // 打开index.html
        if err != nil {
            fmt.Println("文件不存在")
        }
        writer.Write(file) // 将文件内容解析到服务器
    })
    http.ListenAndServe(":9999", nil) // 启动服务器
}

http.Hnadle()

  • 通过Handler接口实现ServeHTTP函数来达到创建处理器
package main

import (
    "io"
    "net/http"
    "time"
)

type TestHandler struct {
}

func (hand *TestHandler) ServeHTTP(respone http.ResponseWriter, request *http.Request) {
    timeNow := time.Now().Format("2006-01-02 15:04:05")
    io.WriteString(respone, timeNow)
}

func main() {
    http.Handle("/time", &TestHandler{})

    http.ListenAndServe(":9999", nil)
}

服务器启动

http.ListenAndServer()

http.ListenAndServe(addr string, handler Handler) error

该函数接收两个参数，一个是字符串第二个是http.Handler

通过使用第一个参数作为地址来启动HTTP服务器（这里用的8080）

请求内容获取

package main

import (
    "fmt"
    "io"
    "net/http"
    "time"
)

type TestHandler struct {
}

func (hand *TestHandler) ServeHTTP(respone http.ResponseWriter, request *http.Request) {
    timeNow := time.Now().Format("2006-01-02 15:04:05")
    io.WriteString(respone, timeNow)
    fmt.Println(request.Method) // 获取请求头的方法
    fmt.Println(request.URL) // 获取请求头的URL
    fmt.Println(request.Proto) // 获取请求头的协议
    fmt.Println(request.Header.Get("User-Agent")) // 获取请求头的UA
    fmt.Println(request.Header.Get("Cookie")) // 获取请求头的Cookie

}

func main() {
    http.Handle("/", &TestHandler{})

    http.ListenAndServe(":9999", nil)
}

参数解析

GET类型

  1. 获取全部参数值
package main

import (
    "fmt"
    "net/http"
)

func main() {
    http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
        request.ParseForm() // 解析参数
        fmt.Println(request.Form) // 会返回一个Map类型的数据，值是一个切片类型，可以存放多个相同参数的值
        fmt.Println(request.Form.Get("user")) // 获取user参数的传参
        fmt.Println(request.Form["user"])     // 获取user参数的传参
    })

    http.ListenAndServe(":9999", nil)
}

/*
map[pass:[123456] user:[admin]]
admin
[admin]
*/
  1. 获取指定参数的值
package main

import (
    "fmt"
    "net/http"
)

func main() {
    http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
        fmt.Println(request.FormValue("user")) // 会自动解析参数，不需要使用 request.ParseForm()
    })

    http.ListenAndServe(":9999", nil)
}

/*
admin
[admin]
*/

POST类型

application/x-www-form-urlencoded

这种编码格式和GET请求类似

  1. request.Form

解析和GET类型的一相同，但是request.Form会将GET传入的参数和POST传入的参数混合再一起，需要使用request.PostForm

http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
    request.ParseForm() // 解析参数
    fmt.Println(request.Form)
})
/*
map[pass:[123456] passwd:[123456] user:[test admin]]
*/

http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
    request.ParseForm() // 解析参数
    fmt.Println(request.PostForm)
})
/*
map[pass:[123456] user:[test]]
*/
  1. request.PostFormValue()
http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
    fmt.Println(request.PostFormValue("user"))
})
/*
test
*/

mutipart/form-data

package main

import (
    "fmt"
    "net/http"
)

func main() {
    http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
        err := request.ParseMultipartForm(1024 * 1024) // 解析提交内容
        if err != nil {
            fmt.Println(err)
        }
        fmt.Println(request.MultipartForm) // 获取数据

    })

    http.ListenAndServe(":9999", nil)
}

获取上传文件内容

request.MultipartForm会返回一个value和file，想要获取上传文件的内容有两种方法

  1. request.MultipartForm.File["a"][0].Open() 会返回file对象
http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
    request.ParseMultipartForm(1024 * 1024) // 解析提交内容

    file, err := request.MultipartForm.File["file"][0].Open() // 获取file传入的数据
    if err != nil {
        fmt.Println(err)
    }
    io.Copy(os.Stdout, file)
})
  1. request.FormFile() 也会返回file对象
http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
    file, _, _ := request.FormFile("file") // 获取file传入的内容
    io.Copy(os.Stdout, file)
})

curl -XPOST "http://127.0.0.1:9999" -F"file=@1.txt"

http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
    file, header, _ := request.FormFile("file")
    io.Copy(os.Stdout, file)
    fmt.Println()
    fmt.Println(header.Filename) // 文件名 
    fmt.Println(header.Size) // 文件大小

解析json内容
package main

import (
    "encoding/json"
    "fmt"
    "net/http"
)

func main() {
    http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
        decoder := json.NewDecoder(request.Body) // 将流传入json解析器中
        var info map[string]interface{} // 创建对应的解析格式
        decoder.Decode(&info) // 解析到指定格式中
        fmt.Println(info) // 输出解析内容
    })

    http.ListenAndServe(":9999", nil)
}

/*
map[a:1]
*/

cookie设置

http.SetCookie

package main

import (
    "fmt"
    "math/rand"
    "net/http"
    "strconv"
    "time"
)

func main() {

    i := rand.New(rand.NewSource(time.Now().Unix()))

    http.HandleFunc("/", func(respone http.ResponseWriter, request *http.Request) {
        cookie := request.Header.Get("cookie")
        fmt.Println(cookie)
        cookies := &http.Cookie{
            Name:     "Guest",
            Value:    strconv.Itoa(i.Int()),
            HttpOnly: true,
        }
        http.SetCookie(respone, cookies)
    })

    http.ListenAndServe(":9999", nil)
}

/*
Guest=3017153804398966264
*/

重定向

http.Redirect

package main

import (
    "fmt"
    "net/http"
)

func main() {
    http.HandleFunc("/home", func(respone http.ResponseWriter, request *http.Request) {
        http.Redirect(respone, request, "/login", 302)
    })

    http.HandleFunc("/login", func(respone http.ResponseWriter, request *http.Request) {
        fmt.Fprint(respone, "登陆界面")
    })

    http.ListenAndServe(":9999", nil)
}

文件服务器

package main

import (
    "net/http"
)

func main() {

    http.Handle("/static/", http.FileServer(http.Dir("."))) // 会在当前目录下找static目录里面的文件

    http.ListenAndServe(":9999", nil)
}

如果想要排除指定目录前缀，可以使用http.StripPrefix去除文件前缀

package main

import (
    "net/http"
)

func main() {

    http.Handle("/static2/", http.StripPrefix("/static2/", http.FileServer(http.Dir("./tmp")))) // 如果不去出前缀，那么回去/static2/tmp去找，因为这里绑定了目录后，会直接通过该绑定目录去寻找，如果去除/static2/，则直接在当前目录的tmp下寻找

    http.ListenAndServe(":9999", nil)
}

POST请求

http.Post()

package main

import (
    "bytes"
    "fmt"
    "io"
    "net/http"
    "os"
)

func main() {

    buffer := bytes.NewBufferString(`{"user": "admin"}`)

    resp, err := http.Post("http://127.0.0.1:9999/", "application/json", buffer)
    if err != nil {
        fmt.Println(err)
    }
    fmt.Println(resp.Request)
    fmt.Println(resp.Status)
    fmt.Println(resp.StatusCode)
    fmt.Println(resp.Proto)

    io.Copy(os.Stdout, resp.Body)

}

/*
200 OK  
200     
HTTP/1.1
*/

此时服务器端会解析传递的json数据 => map[user:admin]

除了直接http.Post()还有一种传输方法http.PostForm()

package main

import (
    "fmt"
    "io"
    "net/http"
    "net/url"
    "os"
)

func main() {

    args := url.Values{}
    args.Add("user", "admin")
    args.Add("user", "test")
    args.Add("pass", "123456")

    resp, err := http.PostForm("http://127.0.0.1:9999/", args)
    if err != nil {
        fmt.Println(err)
    }
    fmt.Println(resp, err)

    fmt.Println(resp.Request)
    fmt.Println(resp.Status)
    fmt.Println(resp.StatusCode)
    fmt.Println(resp.Proto)

    io.Copy(os.Stdout, resp.Body)

}
