网络编程

net包提供了对socket编程的支持，socket变成分为服务端和客户端编程，针对服务端可使用函数Listen创建监听服务，对于客户端可使用Dial连接服务器

Listen			用于创建tcp监听服务器

ListenPacket	用于创建udp服务器端连接

Dial			用于创建于服务器连接

Accept			接收客户端连接

JoinHostPort	连接地址和端口

SplitHostort	分割地址和端口

LookupAddr		查找地址对应主机名

LookupHost		根据主机名查看地址 // 通过本地的host文件以及dns查找

ParseCIDR		解析CIDR格式IP（掩码格式）

package main

import (
    "fmt"
    "net"
)

func main() {
    fmt.Println(net.JoinHostPort("192.168.0.1", "1234")) // 192.168.0.1:1234
    fmt.Println(net.JoinHostPort("::1", "1234"))         // [::1]:1234

    fmt.Println(net.SplitHostPort("192.168.0.1:1234")) // 192.168.0.1 1234 <nil>
    fmt.Println(net.SplitHostPort("[::1]:1234"))       // ::1 1234 <nil>

    fmt.Println(net.LookupAddr("127.0.0.1"))     // [DESKTOP-M8HMTLF] <nil>
    fmt.Println(net.LookupHost("www.baidu.com")) // [110.242.68.3 110.242.68.4] <nil>
}

常用结构体

IP

常用函数

ParseIP : 解析地址（解析成IP结构体格式）

LookupIP : 根据主机名查找IP

package main

import (
    "fmt"
    "net"
)

func main() {
    ip := net.ParseIP("127.0.0.1")
    // net.IP => net.IP{0x0, 0x0, 0x0, 0x0, 0x0, 0x0, 0x0, 0x0, 0x0, 0x0, 0xff, 0xff, 0x7f, 0x0, 0x0, 0x1}
    fmt.Printf("%T => %#v\n", ip, ip)

    lookupIP, err := net.LookupIP("www.baidu.com")
    if err != nil {
        fmt.Println(err)
    }
    // []net.IP => []net.IP{net.IP{0x6e, 0xf2, 0x44, 0x3}, net.IP{0x6e, 0xf2, 0x44, 0x4}}
    fmt.Printf("%T => %#v\n", lookupIP, lookupIP)
}

IPNet

IP范围格式

常用方法

Contains : 判断是否包含某IP

NetWork : 获取网络地址类型

package main

import (
    "fmt"
    "net"
)

func main() {
    ip := net.ParseIP("192.168.1.100")

    cidr, ipNet, err := net.ParseCIDR("192.168.1.10/24")
    fmt.Println(cidr, ipNet, err) // 192.168.1.10 192.168.1.0/24 <nil>

    fmt.Println(ipNet.Contains(ip)) // true
}

Interface

常用属性

Index :

MTU : 最大传输单元

Name : 网卡名

Flags : 属性

常用函数

Interfaces : 获取所有网卡

常用方法

Addrs : 获取网卡配置网络地址

package main

import (
    "fmt"
    "net"
)

func main() {
    info, err := net.Interfaces()
    if err != nil {
        fmt.Println(err)
    }
    for _, v := range info {
        fmt.Println("-------------------------------------------")
        fmt.Println(v.Name, v.MTU, v.HardwareAddr, v.Flags, v.Index)
        ips, _ := v.Addrs()
        for _, ips := range ips {
            fmt.Println(ips)
        }
        fmt.Println("-------------------------------------------")
    }
}

简易网络通讯

tcp

服务端

客户端发送消息 => 从控制台输入

服务器回复消息 => 从控制台输入

package main

import (
    "bufio"
    "fmt"
    "net"
)

func main() {
    listen, err := net.Listen("tcp", "127.0.0.1:1234") // 创建tcp监听
    if err != nil {
        fmt.Println(err)
    }
    conn, err := listen.Accept() // 接收客户端请求
    if err != nil {
        fmt.Println(err)
    }
    reader := bufio.NewReader(conn) // 读取客户端发送的数据

    func() {
        defer conn.Close()
        fmt.Printf("客户端 [%s] 连接成功\n", conn.RemoteAddr()) // 显示哪个客户端进行连接

        for {
            line, _, err := reader.ReadLine() // 读取一行数据

            if err != nil || string(line) == "quit" { // 当err为空或者输入quit时退出
                fmt.Printf("客户端 [%s] 断开连接\n", conn.RemoteAddr()) // 打印客户端断开连接
                fmt.Fprintf(conn, "%s 断开连接\n", conn.RemoteAddr()) // 当客户端断开连接时发送客户端数据
                break
            } else {
                fmt.Println(string(line))
            }
        }
    }()
}
/*
客户端 [127.0.0.1:12909] 连接成功
这是第 1 条数据                  
这是第 2 条数据                  
这是第 3 条数据                  
这是第 4 条数据                  
客户端 [127.0.0.1:12909] 断开连接
*/

客户端

package main

import (
    "bufio"
    "fmt"
    "net"
)

func main() {
    ip, err := net.Dial("tcp", "127.0.0.1:1234") // 连接服务端
    if err != nil {
        fmt.Println(err)
    }
    defer ip.Close() // 延迟关闭

    reader := bufio.NewReader(ip) // 读取服务器发送的数据
    for i := 0; i < 4; i++ {
        fmt.Fprintf(ip, "这是第 %d 条数据\n", i+1) // 给服务器发送数据
    }
    fmt.Fprintln(ip, "quit") // 给服务端发送quit指令
    line, _, _ := reader.ReadLine() // 输出服务端给客户端的数据
    fmt.Println(string(line))
}
/*
127.0.0.1:12909 断开连接
*/

udp

服务端

package main

import (
    "fmt"
    "net"
)

func main() {
    address := ":9900"
    packet, err := net.ListenPacket("udp", address) //创建udp连接
    if err != nil {
        fmt.Println(err)
    }
    defer packet.Close() // 延迟关闭连接

    ctx := make([]byte, 1024)

    n, addr, err := packet.ReadFrom(ctx) // 读取来自客户端的数据，并返回数据长度、客户端信息、以及错误信息

    if err != nil {
        fmt.Println(err)
    }

    fmt.Println(addr, " : ", string(ctx[:n]))

    packet.WriteTo([]byte("发送给客户端的数据"), addr) // 发送给连接客户端信息

}

/*
127.0.0.1:62337  :  time : 2023-09-07 21:26:08
*/

客户端

package main

import (
    "bufio"
    "fmt"
    "net"
    "time"
)

func main() {

    ctx := make([]byte, 1024)
    conn, err := net.Dial("udp", "127.0.0.1:9900") // 创建udp连接

    if err != nil {
        fmt.Println(err)
    }
    defer conn.Close() // 延迟关闭

    fmt.Fprintf(conn, "time : %s", time.Now().Format("2006-01-02 15:04:05")) // 给服务端发送数据

    reader := bufio.NewReader(conn) // 读取服务端数据

    n, err := reader.Read(ctx) 
    if err != nil {
        fmt.Println(err)
    }
    fmt.Println(string(ctx[:n]))
}

发送给客户端的数据
