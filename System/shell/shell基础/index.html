shell变量

变量的作用

为灵活管理linux系统提供特定参数，由两层意思

  • 变量名 : 使用固定的名称，由系统预设或用户定义
  • 变量值 : 能够根据用户设置、系统环境的变化而变化

变量的类型

  • 自定义变量 : 由用户自定义、使用和修改
a=123
echo $a
  • 环境变量 : 由系统维护，用于设置工作环境
echo $PATH
  • 位置变量 : 通过命令行给脚本程序传递参数
  • 预定义变量 : Bash中内置的一类变量，不能直接修改
echo $?

变量的定义

定义一个新的变量

  • 变量名以字母或下划线开头，区分大小写，建议全大写

变量名=变量值

查看一个变量的值

echo $变量名

赋值引号

  1. 单引号

    不允许引用其他变量的值，$为普通字符

  2. 双引号

    允许通过$引用其他变量的值

  3. 反撇号

    命令替换、提取命令执行后的输出结果

  4. $()

    和反撇号类似

a1="11"
a2="22"
a3="$a1,$a2,33"

echo $a3 // 輸出11,22,33

a3 = '$a1,$a2,33'
echo $a3 // 輸出$a1,$a2,33

a3=`ls`
echo $a3 // 命令執行結果就放到a3

a3=$(ls)
echo $a3

正则表达式

基础

  1. 正则表达式定义

正则表达式，又称正规表达式、常规表达式，是使用单个字符来描述、匹配一系列符合某个句法规则的字符串

  1. 正则表达式组成

普通字符

  • 大小写字母、数字、标点符号及一些其他符号

元字符

  • 在正则表达式中具有特殊意义的专用字符
  1. 常见元字符
^			以什么开头 ^help（以help开头）
$			以什么结尾 help$（以help结尾）
.			匹配单个任意字符			
*			匹配*左边的表达式0次到任意次
+			匹配+左边的表达式1次到任意次0次或1次
?			匹配?左边的表达式
\			使右侧正则符号失去含义变为字符
[]			范围选择匹配符（匹配[]里面的）
[^]			取反（范围选择中的取反）[^a-zA-Z] 除去a-z和A-Z其他都匹配
[n1|n2]		"|"多个信息中匹配某一个
[n1-n2]		"-"匹配一个范围
\{n\}		匹配左侧表达式的个数
\{n,\}		和 * 效果一样
\{n,m\}		匹配左侧表达式的n到m次
()			查找组字符串
()+			辨别多个重复的组

sed

a : 添加数据

  1. 每行都添加数据（不加$是每一行下面都会添加信息）

sed -i ‘a 要添加的信息’ 要添加信息的文件

sed -i ‘a 123’ /etc/sysconfig/network-scripts/ifcfg-ens33

  1. 文件末尾添加（加$末尾行下添加信息）

sed -i ‘$a 要添加的信息’ 要添加信息的文件

sed -i ‘$a IPADDR=192.168.50.123’ /etc/sysconfig/network-scripts/ifcfg-ens33

  1. 指定某行 添加

sed -i ‘/规则/ a 添加的信息’ 添加信息的文件

// 匹配以ONBOOT开头的下一行插入信息

sed -i ‘/^ONBOOT/ a IPADDR=192.168.50.123’ /etc/sysconfig/network-scripts/ifcfg-ens33

s : 匹配数据

  1. 匹配某些行进行修改

sed -i ‘/规则/s/原先的数据/替换的数据/g’ 要匹配的文件

sed -i ‘/^ONBOOT/s/no/yes/g’ /etc/sysconfig/network-scripts/ifcfg-ens33

  1. 直接替换

sed -i ‘s/原先的数据/替换的数据/g’ 要匹配的文件

sed -i ‘s/ONBOOT=no/ONBOOT=yes/g’ /etc/sysconfig/network-scripts/ifcfg-ens33

  1. 匹配某些行进行注释

sed -i ‘s/要注释的行特征/#&/g’ 要匹配的文件

sed -i ‘s/UUID/$&/g’ /etc/sysconfig/network-scripts/ifcfg-ens33

d : 删除某些行

sed -i ‘/规则/d’ 要匹配的文件

sed -i ‘/^IPADDR/d’ /etc/sysconfig/network-scripts/ifcfg-ens33

一键安装dhcp

#! /bin/bash

if [ -e /media/Packages ]
then
echo "centos7已经挂载"
else
umount /dev/sr0
mount /dev/sr0 /meida
fi

if [ -e /etc/yum.repos.d/local.repo]
then
echo "local.repo已存在"
else
rm -rf /etc/yum.repos.d/*
echo "[local-yum]" >> /etc/yum.repos.d/local.repo
echo "name=local-yum"  >> /etc/yum.repos.d/local.repo
echo "baseurl=file:///media/" >> /etc/yum.repos.d/local.repo
echo "enable=1" >> /etc/yum.repos.d/local.repo
echo "gpgcheck=0" >> /etc/yum.repos.d/local.repo
yum list
fi

rpm -q dhcpd
if [ $? -eq 0]
then
echo "已安装dhcp"
else
yum -y install dhcp*
fi


read -p "输入网络地址" NET
read -p "输入子网掩码" MASK
read -p "输入起始IP" STARTIP
read -p "输入结束IP" STOPIP
read -p "输入DNS地址(多个IP可以使用逗号分割)" DNS
read -p "输入网关" GATE
read -p "输入广播地址" BROA

echo "subnet $NET netmask $MASK {" > /etc/dhcp/dhcpd.conf
echo "  range $STARTIP $STOPIP;" >> /etc/dhcp/dhcpd.conf
echo "  option domain-name-servers $DNS" >> /etc/dhcp/dhcpd.conf
echo '  option domain-name "internal.example.org"; ' >> /etc/dhcp/dhcpd.conf
echo "  option routers $GATE; " >> /etc/dhcp/dhcpd.conf
echo "  option broadcast-address $BROA; " >> /etc/dhcp/dhcpd.conf
echo "  default-lease-time 600;" >> /etc/dhcp/dhcpd.conf
echo "  max-lease-time 7200;" >> /etc/dhcp/dhcpd.conf
echo "}" >> /etc/dhcp/dhcpd.conf

systemctl start dhcpd &> /dev/null
if [ $? -qe 0 ]
then
echo "成功启动"
else
systemctl status dhcpd
fi

#!/bin/bash
echo "---------------YUM Server Config -------------------"
umount /dev/sr0
mount /dev/sr0 /meida
rm -rf /etc/yum.repos.d/*
echo "[test]">>/etc/yum.repos.d/test.repo

echo "enabled=1">>/etc/yum.repos.d/test.repo
echo "gpgcheck=1">>/etc/yum.repos.d/test.repo

read -p "public dir" : DIR
read -p "username : " USER
mkdir -p $DIR
yum -y install samba*
useradd $USER
echo -e "123.com\n123.com" | pdbedit -u -a -t $USER
sed -i '$a \['$DIR'\]'/etc/smaba/smb.conf
sed -i '$a path='$DIR''/etc/smaba/smb.conf
sed -i '$a public=no'/etc/smaba/smb.conf
sed -i '$a read only=yes'/etc/smaba/smb.conf
sed -i '/^\[global\]$/ a username map=/etc/smaba/smbusers' /etc/samba/smb.conf

一键配置IP

#!/bin/bash
read -p "ip address :"  IP
echo $IP | grep "^[0-9]\{1,3\}\.[0-9]\{1,3\}\.[0-9]\{1,3\}\.[0-9]\{1,3\}$" &> /dev/null
if [ $? -eq 0 ]
then
A=`echo $IP | awk -F. '{print $1}'`
B=`echo $IP | awk -F. '{print $2}'`
C=`echo $IP | awk -F. '{print $3}'`
D=`echo $IP | awk -F. '{print $4}'`
    if [ $A -ge 1 ] && [ $A -le 255 ]
    then
    echo A is OK
    else
    echo A is Error
    exit 1
    fi
    if [ $B -ge 1 ] &&  [ $B -le 255 ]
    then
    echo B is OK
    else
    echo B is Error
    exit 1
    fi
    if [ $C -ge 1 ] && [ $C -le 255 ]
    then
    echo C is OK
    else
    echo C is Error
    exit 1
    fi
    if [ $D -ge 1 ] && [ $D -le 255 ]
    then
    echo D is OK
    else
    echo D is Error
    exit 1
    fi
else
echo 000.000.000.000-255.255.255.255
fi

LAN=`ifconfig | head -1 |awk -F : '{print $1}'`

sed -i "/^ONBOOT/s/no/yes/g" /etc/sysconfig/network-scripts/ifcfg-$LAN
sed -i "/^BOOTPROTO/s/dhcp/static/g" /etc/sysconfig/network-scripts/ifcfg-$LAN
sed -i "s/^UUID/#&/g" /etc/sysconfig/network-scripts/ifcfg-$LAN

if [ $A -ge 1 ] && [ $A -le 126 ]
then
MASK='255.0.0.0'
elif [ $A -ge 126 ] && [ $A -le 191 ]
then
MASK='255.255.0.0'
elif [ $A -ge 191 ] && [ $A -le 223 ]
then
MASK='255.255.255.0'
else
echo Error
fi
grep -ni "IPADDR" /etc/sysconfig/network-scripts/ifcfg-$LAN >& /dev/null
if [ $? -eq 0 ]
then
sed -i "s/IPADDR=*.*.*.*/IPADDR=$IP/g" /etc/sysconfig/network-scripts/ifcfg-$LAN
sed -i "s/NETMASK=*.*.*.*/NETMASK=$MASK/g" /etc/sysconfig/network-scripts/ifcfg-$LAN
else
echo "IPADDR=$IP" >> /etc/sysconfig/network-scripts/ifcfg-$LAN
echo "NETMASK=$MASK" >> /etc/sysconfig/network-scripts/ifcfg-$LAN
fi
systemctl restart network
if [ $? -eq 0 ]
then
echo service is ok
else
echo `systemctl status network`
fi
ifconfig | grep $IP >& /dev/nul
if [ $? -eq 0 ]
then
echo ip is true
else
echo ip is false
fi

猜数字

#! /bin/bash

i=0
randoms=`expr $RANDOM % 1000`
while true
    do
        read -p "输入数字 : " NUM
        let i++
        if [ $NUM -eq $randoms ]
            then
            echo "猜对了，一共猜了$i次"
            exit 0
        elif [ $NUM -gt $randoms ]
            then
            echo 大了
        else
            echo 小了
        fi
    done
