# 实验拓扑

# area0 区域配置

# AR1 配置

  1. 配置 IP 地址
[AR1]interface GigabitEthernet 0/0/0
[AR1-GigabitEthernet0/0/0]ip address 192.168.12.1 24
[AR1]interface GigabitEthernet 0/0/1
[AR1-GigabitEthernet0/0/1]ip address 172.16.20.1 24
[AR1]interface GigabitEthernet 1/0/0
[AR1-GigabitEthernet1/0/0]ip address 192.168.13.1 24
[AR1]interface LoopBack 0
[AR1-LoopBack0]ip address 1.1.1.1 24
  1. 配置 ospf-area0 区域
[AR1]ospf router-id 1.1.1.1
[AR1-ospf-1]area 0
[AR1-ospf-1-area-0.0.0.0]network 172.16.20.0 0.0.0.255
[AR1-ospf-1-area-0.0.0.0]network 192.168.12.0 0.0.0.255
[AR1-ospf-1-area-0.0.0.0]network 192.168.13.0 0.0.0.255
[AR1-ospf-1-area-0.0.0.0]network 1.1.1.0 0.0.0.255

# area1 区域配置

# SW2 配置

  1. e0/0/4 接口连接 PC 配置
[SW2]vlan batch 10 20
[SW2]interface Ethernet0/0/4
[SW2-Ethernet0/0/4]port link-type access 
[SW2-Ethernet0/0/4]port default vlan 10
  1. e0/0/5 接口连接 PC 配置
[SW2]interface Ethernet0/0/5
[SW2-Ethernet0/0/5]port link-type access
[SW2-Ethernet0/0/5]port default vlan 20
  1. e0/0/1 to e0/0/3 配置聚合链路（<font color="#2DC26B"> 这里 SW1 为主动端，SW2 为被动端 </font>）
[SW2]int Eth-Trunk 1
[SW2-Eth-Trunk1]mode lacp-static 
[SW2-Eth-Trunk1]trunkport Ethernet 0/0/1 to 0/0/3
[SW2-Eth-Trunk1]port link-type trunk 
[SW2-Eth-Trunk1]port trunk allow-pass vlan 10 20

# SW1 配置

[SW1]vlan batch 10 12 20
[SW1]dhcp enable
  1. vlan10 配置 PC 网关、DHCP 服务
[SW1]interface Vlanif 10
[SW1-Vlanif10]ip address 192.168.10.1 24
[SW1-Vlanif10]dhcp select interface 
[SW1-Vlanif10]dhcp server excluded-ip-address 192.168.10.230 192.168.10.254	
[SW1-Vlanif10]dhcp server dns-list 8.8.8.8
  1. vlan20 配置 PC 网关、DHCP 服务
[SW1]interface Vlanif 20
[SW1-Vlanif20]ip address 192.168.20.1 24
[SW1-Vlanif20]dhcp select interface
[SW1-Vlanif20]dhcp server excluded-ip-address 192.168.20.230 192.168.20.254
[SW1-Vlanif20]dhcp server dns-list 8.8.8.8
  1. g0/0/4 配置出接口类型为 access
[SW1]interface GigabitEthernet0/0/4
[SW1-GigabitEthernet0/0/4]port link-type access
[SW1-GigabitEthernet0/0/4]port default vlan 12
[SW1]int Vlanif 12
[SW1-Vlanif12]ip address 192.168.12.100 24
  1. 配置 LoopBack 口
[SW1]interface LoopBack 0
[SW1-LoopBack0]ip address 2.2.2.2 24
  1. 配置 ospf-area0 和 area1 区域
[SW1]ospf router-id 2.2.2.2
[SW1-ospf-1]area 0
[SW1-ospf-1-area-0.0.0.0]network 192.168.12.0 0.0.0.255
[SW1-ospf-1-area-0.0.0.0]network 2.2.2.0 0.0.0.255
[SW1-ospf-1]area 1
[SW1-ospf-1-area-0.0.0.1]network 192.168.10.0 0.0.0.255
[SW1-ospf-1-area-0.0.0.1]network 192.168.20.0 0.0.0.255



4. g0/0/1 to g0/0/3 配置聚合链路（<font color="#2DC26B"> 这里 SW1 为主动端，SW2 为被动端 </font>）

[SW1]lacp priority 0
[SW1]int Eth-Trunk 1	
[SW1-Eth-Trunk1]mode lacp-static 
[SW1-Eth-Trunk1]trunkport GigabitEthernet 0/0/1 to 0/0/3
[SW1-Eth-Trunk1]max active-linknumber 2
[SW1-Eth-Trunk1]port link-type trunk 
[SW1-Eth-Trunk1]port trunk allow-pass vlan 10 20

# area2 区域配置

# SW3 配置

  1. e0/0/1 接口连接 PC 配置
[SW3]interface Ethernet0/0/1
[SW3-Ethernet0/0/1]port link-type access 
[SW3-Ethernet0/0/1]port default vlan 30
  1. e0/0/2 接口连接 PC 配置
[SW3]interface Ethernet0/0/2
[SW3-Ethernet0/0/2]port link-type access
[SW3-Ethernet0/0/2]port default vlan 40
  1. g0/0/1 接口连接交换机配置
[SW3]interface GigabitEthernet 0/0/1
[SW3-GigabitEthernet0/0/1]port link-type trunk 
[SW3-GigabitEthernet0/0/1]port trunk allow-pass vlan 30 40

# AR2 配置

[AR2]dhcp enable 
[AR2]int LoopBack 0
[AR2-LoopBack0]ip address 3.3.3.3 24
  1. g0/0/0 接口配置 IP
[AR2]interface GigabitEthernet 0/0/0
[AR2-GigabitEthernet0/0/0]ip address 192.168.13.100 24
  1. g0/0/1 接口配置单臂路由 DHCP
[AR2]interface GigabitEthernet0/0/1.30
[AR2-GigabitEthernet0/0/1.30]dot1q termination vid 30
[AR2-GigabitEthernet0/0/1.30]ip address 192.168.30.1 255.255.255.0 
[AR2-GigabitEthernet0/0/1.30]arp broadcast enable
[AR2-GigabitEthernet0/0/1.30]dhcp select interface
[AR2-GigabitEthernet0/0/1.30]dhcp server dns-list 8.8.8.8 
[AR2]interface GigabitEthernet 0/0/1.40
[AR2-GigabitEthernet0/0/1.40]ip address 192.168.40.1 24
[AR2-GigabitEthernet0/0/1.40]dot1q termination vid 40
[AR2-GigabitEthernet0/0/1.40]arp broadcast enable
[AR2-GigabitEthernet0/0/1.40]dhcp select interface 
[AR2-GigabitEthernet0/0/1.40]dhcp server dns-list 8.8.8.8
  1. 配置 ospf-area0 和 area2 区域
[AR2]ospf router-id 3.3.3.3
[AR2-ospf-1]area 0
[AR2-ospf-1-area-0.0.0.0]network 192.168.13.0 0.0.0.255
[AR2-ospf-1-area-0.0.0.0]network 3.3.3.0 0.0.0.255
[AR2-ospf-1]area 2
[AR2-ospf-1-area-0.0.0.2]network 192.168.30.0 0.0.0.255
[AR2-ospf-1-area-0.0.0.2]network 192.168.40.0 0.0.0.255


# 阻止 Client1 访问 Server1

AR1 路由器配置

[AR1]acl number 3000
[AR1-acl-adv-3000]rule deny tcp source 192.168.30.0 0.0.0.255 destination 172.16.20.100 0 destination-port eq 80
[AR1]interface GigabitEthernet 0/0/1	
[AR1-GigabitEthernet0/0/1]traffic-filter outbound acl 3000

# 配置 PPPoE 服务器

  1. ISP 路由器配置
[ISP]ip pool 200_1_1
[ISP-ip-pool-200_1_1]network 200.1.1.0 mask 24
[ISP-ip-pool-200_1_1]gateway-list 200.1.1.1
[ISP-ip-pool-200_1_1]dns-list 114.114.114.114
[ISP]int Virtual-Template 1
[ISP-Virtual-Template1]ppp authentication-mode chap
[ISP-Virtual-Template1]remote address pool 200_1_1
[ISP-Virtual-Template1]ip address 200.1.1.1 24
[ISP]interface GigabitEthernet 0/0/2
[ISP-GigabitEthernet0/0/2]pppoe-server bind virtual-template 1
[ISP]aaa
[ISP-aaa]local-user mpy password cipher 123.com 
[ISP-aaa]local-user mpy service-type ppp
  1. 配置 LoopBack 口
[ISP]interface LoopBack 0
[ISP-LoopBack0]ip address 8.8.8.8 24

# 配置 PPPoE 客户端

  1. AR1 路由器配置
[AR1]interface Dialer 1
[AR1-Dialer1]link-protocol ppp
[AR1-Dialer1]dialer user mpy
[AR1-Dialer1]dialer bundle 1
[AR1-Dialer1]ppp chap user mpy
[AR1-Dialer1]ppp chap password cipher 123.com
[AR1-Dialer1]ip address ppp-negotiate 
[AR1]interface GigabitEthernet 0/0/2
[AR1-GigabitEthernet0/0/2]pppoe-client dial-bundle-number 1


2. 配置访问外网的路由和策略

[AR1]acl number 2000
[AR1-acl-basic-2000]rule permit source any 
[AR1]interface Dialer 1
[AR1-Dialer1]nat outbound 2000
 nat server protocol tcp global 200.1.1.2 www inside 172.16.20.100 www
[AR1]ip route-static 0.0.0.0 0 Dialer 1
[AR2]ip route-static 0.0.0.0 0 192.168.13.1
[SW1]ip route-static 0.0.0.0 0 192.168.12.1
