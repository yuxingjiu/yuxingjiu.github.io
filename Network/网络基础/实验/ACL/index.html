ACL

配置指令

SW1配置

[SW1]vlan batch 10 20 100
[SW1]ip route-static 0.0.0.0 0 192.168.0.2

[SW1]dhcp enable

[SW1]interface GigabitEthernet0/0/1
[SW1-GigabitEthernet0/0/1]port link-type trunk
[SW1-GigabitEthernet0/0/1]port trunk allow-pass vlan 10
[SW1]ip pool 10_1_1
[SW1-ip-pool-10_1_1]gateway-list 10.1.1.1
[SW1-ip-pool-10_1_1]network 10.1.1.0 mask 255.255.255.0
[SW1-ip-pool-10_1_1]dns-list 8.8.8.8
[SW1]interface Vlanif10
[SW1-Vlanif10]ip address 10.1.1.1 255.255.255.0
[SW1-Vlanif10]dhcp select global

[SW1]interface GigabitEthernet0/0/3
[SW1-GigabitEthernet0/0/3]port link-type trunk
[SW1-GigabitEthernet0/0/3]port trunk allow-pass vlan 20
[SW1]ip pool 10_1_2
[SW1-ip-pool-10_1_2]network 10.1.2.0 mask 24
[SW1-ip-pool-10_1_2]gateway-list 10.1.2.1	
[SW1-ip-pool-10_1_2]dns-list 8.8.8.8
[SW1]interface Vlanif20
[SW1-Vlanif20]ip address 10.1.2.1 255.255.255.0
[SW1-Vlanif20]dhcp select global 

[SW1]interface GigabitEthernet0/0/4
[SW1-GigabitEthernet0/0/4]port link-type access
[SW1-GigabitEthernet0/0/4]port default vlan 100
[SW1]interface Vlanif100
[SW1-Vlanif100]ip address 192.168.0.1 255.255.255.252

SW2配置

[SW2]vlan 10
[SW2]interface Ethernet0/0/1
[SW2-Ethernet0/0/1]port link-type access
[SW2-Ethernet0/0/1]port default vlan 10
[SW2]interface Ethernet0/0/2
[SW2-Ethernet0/0/2]port link-type access
[SW2-Ethernet0/0/2]port default vlan 10
[SW2]interface GigabitEthernet0/0/1
[SW2-GigabitEthernet0/0/1]port link-type trunk
[SW2-GigabitEthernet0/0/1]port trunk allow-pass vlan 10

SW3配置

[SW3]vlan 20
[SW3]interface GigabitEthernet0/0/1
[SW3-GigabitEthernet0/0/1] port link-type trunk
[SW3-GigabitEthernet0/0/1]port trunk allow-pass vlan 20
[SW3]interface Ethernet0/0/1
[SW3-Ethernet0/0/1]port link-type access
[SW3-Ethernet0/0/1]port default vlan 20
[SW3]interface Ethernet0/0/2
[SW3-Ethernet0/0/2]port link-type access
[SW3-Ethernet0/0/2]port default vlan 20

AR1配置

[AR1]interface GigabitEthernet0/0/0
[AR1-GigabitEthernet0/0/0]ip address 192.168.0.2 255.255.255.252 
[AR1]interface GigabitEthernet0/0/1
[AR1-GigabitEthernet0/0/1]ip address 50.50.50.1 24
[AR1]interface GigabitEthernet0/0/2
[AR1-GigabitEthernet0/0/2]ip address 100.200.1.1 24
[AR1]ip route-static 10.1.1.0 24 192.168.0.1
[AR1]ip route-static 10.1.2.0 24 192.168.0.1

配置策略

[AR1]acl number 2000
[AR1-acl-basic-2000]rule deny source 10.1.2.0 0.0.0.255
[AR1-acl-basic-2000]rule permit source any  
[AR1-GigabitEthernet0/0/2]traffic-filter outbound acl 2000
[AR1]acl number 3000
[AR1-acl-adv-3000]rule deny tcp source 10.1.1.3 0 destination 100.200.1.2 0 destination-port eq ftp
[AR1-acl-adv-3000]rule deny icmp source 10.1.1.0 0.0.0.255 destination 50.50.50.2 0 
[AR1-acl-adv-3000]rule permit ip source 10.1.2.254 0
[AR1-acl-adv-3000]rule deny ip source 10.1.2.0 0.0.0.255
[AR1]interface GigabitEthernet0/0/0
[AR1-GigabitEthernet0/0/0]traffic-filter inbound acl 3000
